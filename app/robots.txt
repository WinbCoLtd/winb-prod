import { NextResponse } from "next/server";

export async function GET() {

User-agent: *
Disallow: /admin/


Allow: /$
Allow: /en/
Allow: /ja/
Allow: /companyPrifile/
Allow: /vehicleList/
Allow: /otherService/
Allow: /contact/

  return new NextResponse(robotsTxt, {
    headers: {
      "Content-Type": "text/plain",
    },
  });
}
